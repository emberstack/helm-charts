To connect to Redis:

{{- if .Values.auth.enabled }}

  1. Get the Redis password:

     export REDIS_PASSWORD=$(kubectl get secret {{ include "redis.secretName" . }} -n {{ include "redis.namespace" . }} -o jsonpath="{.data.{{ include "redis.secretKey" . }}}" | base64 -d)

{{- end }}

  {{ if .Values.auth.enabled }}2{{ else }}1{{ end }}. Connect using redis-cli:

{{- if eq .Values.architecture "standalone" }}

     kubectl run redis-cli --rm -it --restart=Never -n {{ include "redis.namespace" . }} \
       --image={{ include "redis.image" . }} -- \
       redis-cli -h {{ include "redis.fullname" . }}{{ if .Values.auth.enabled }} -a "$REDIS_PASSWORD"{{ end }}{{ include "redis.tls.cliArgs" . }}

{{- else if and (eq .Values.architecture "replication") .Values.sentinel.enabled }}

     # Connect via Sentinel (discovers the current master automatically):
     kubectl run redis-cli --rm -it --restart=Never -n {{ include "redis.namespace" . }} \
       --image={{ include "redis.image" . }} -- \
       redis-cli -h {{ include "redis.fullname" . }} -p {{ .Values.sentinel.port }}{{ if .Values.auth.enabled }} -a "$REDIS_PASSWORD"{{ end }} sentinel get-master-addr-by-name {{ .Values.sentinel.masterName }}

{{- else }}

     kubectl run redis-cli --rm -it --restart=Never -n {{ include "redis.namespace" . }} \
       --image={{ include "redis.image" . }} -- \
       redis-cli -h {{ include "redis.fullname" . }}{{ if .Values.auth.enabled }} -a "$REDIS_PASSWORD"{{ end }}{{ include "redis.tls.cliArgs" . }}

{{- end }}
